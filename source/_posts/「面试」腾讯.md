```java
int strlen( const char * s){
	
	if(s == null){
		return 0;
	}

	int len = 0;
	while(s[len] != '\0'){
		len++;
	}

	return len;

}

int strlen(const char * s){
	
	return  s == null ? 0 : strlen( s+1) + 1; 
}
```
